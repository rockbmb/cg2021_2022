128void spherical2Cartesian (double radius, double elevation, double azimuth, double *x, double *y, double *z)
129{
130
131 *x = radius * cos (elevation) * sin (azimuth) + globalCenterX;
132 *y = radius * sin (elevation) + globalCenterY;
133 *z = radius * cos (elevation) * cos (azimuth) + globalCenterZ;
134}
135
136void cartesian2Spherical (double x, double y, double z, double *radius, double *azimuth, double *elevation)
137{
138 *radius = abs (sqrt (pow (x, 2) + pow (y, 2) + pow (z, 2)));
139 *elevation = asin (y / (*radius));
140 *azimuth = asin (x / (*radius * cos (*elevation)));
141}

209Model allocModel (const char *path)
210{
211 FILE *fp = fopen (path, "r");
212 if (!fp)
213 {
214 fprintf (stderr, "failed to open model: %s\n", path);
215 exit (1);
216 }
217
218 unsigned int nVertices;
219 fread (&nVertices, sizeof (unsigned int), 1, fp);
220 float *modelBuf = (float *) malloc (3 * nVertices * sizeof (float));
221
222 fread (modelBuf, 3 * sizeof (float), nVertices, fp);
223 fclose (fp);
224
225 Model model = (Model) malloc (sizeof (Model));
226 if (model == NULL)
227 {
228 fprintf (stderr, "Failed loading model\n");
229 exit (1);
230 }
231
232 model->nVertices = nVertices;
233 model->vertices = modelBuf;
234
235 return model;
236}

282void changeSize (int w, int h)
283{
284
285 // Prevent a divide by zero, when window is too short
286 // (you cant make a window with zero width).
287 if (h == 0)
288 h = 1;

291 float ratio = (float) w * 1.0f / (float) h;
292
293 // Set the projection matrix as current
294 glMatrixMode (GL_PROJECTION);
295 // Load Identity Matrix
296 glLoadIdentity ();
297
298 // Set the viewport to be the entire window
299 glViewport (0, 0, w, h);
300
301 // Set perspective
302 gluPerspective (globalFOV, ratio, globalNear, globalFar); //fox,near,far

317void operations_render (std::vector<float> *operations)
318{
319 unsigned int i = 0;
320 static bool hasPushedModels = false;
321 static bool hasLoadedCurves = false;
322 static vector<vector<vec3>> curves;
323
324 DEFAULT_GLOBAL_EYE_X = operations->at (i);
325 DEFAULT_GLOBAL_EYE_Y = operations->at (i + 1);
326 DEFAULT_GLOBAL_EYE_Z = operations->at (i + 2);
327
328 cartesian2Spherical (
329 DEFAULT_GLOBAL_EYE_X, DEFAULT_GLOBAL_EYE_Y, DEFAULT_GLOBAL_EYE_Z,
330 &DEFAULT_GLOBAL_RADIUS, &DEFAULT_GLOBAL_AZIMUTH, &DEFAULT_GLOBAL_ELEVATION);
331 i += 3;
332
333 DEFAULT_GLOBAL_CENTER_X = operations->at (i);
334 DEFAULT_GLOBAL_CENTER_Y = operations->at (i + 1);
335 DEFAULT_GLOBAL_CENTER_Z = operations->at (i + 2);
336 i += 3;
337
338 DEFAULT_GLOBAL_UP_X = operations->at (i);
339 DEFAULT_GLOBAL_UP_Y = operations->at (i + 1);
340 DEFAULT_GLOBAL_UP_Z = operations->at (i + 2);
341 i += 3;
342
343 DEFAULT_GLOBAL_FOV = operations->at (i);
344 DEFAULT_GLOBAL_NEAR = operations->at (i + 1);
345 DEFAULT_GLOBAL_FAR = operations->at (i + 2);
346
347 unsigned int model_num = 0;
348
349 for (i += 3; i < operations->size (); i++)
350 {
351 switch ((int) operations->at (i))
352 {
353 case ROTATE:
354 glRotatef (operations->at (i + 1),
355 operations->at (i + 2),
356 operations->at (i + 3),
357 operations->at (i + 4));
358 i += 4;
359 continue;
360 case EXTENDED_ROTATE:
361 {
362 advance_in_rotation (
363 operations->at (i + 1),
364 {
365 operations->at (i + 2),
366 operations->at (i + 3),
367 operations->at (i + 4)
368 });
369 i += 4; //time, axis_of_rotation
370 continue;
371 }
372 case TRANSLATE:
373 glTranslatef (operations->at (i + 1),
374 operations->at (i + 2),
375 operations->at (i + 3));
376 i += 3;
377 continue;
378 case EXTENDED_TRANSLATE:
379 {
380 int number_of_points = (int) operations->at (i + 3);
381 vector<vec3> new_curve (number_of_points);
382 if (!hasLoadedCurves)
383 {
384 for (int j = 0; j < number_of_points; ++j)
385 {
386 int idx = 3*j;
387 new_curve.at (j)[0] = operations->at (i + 4 + idx);
388 new_curve.at (j)[1] = operations->at (i + 4 + idx + 1);
389 new_curve.at (j)[2] = operations->at (i + 4 + idx + 2);
390 }
391 curves.push_back (new_curve);
392 }
393 auto curve = curves.back ();
394 renderCurve (Mcr, curve);
395 advance_in_curve (
396 operations->at (i + 1),
397 (bool) operations->at (i + 2),
398 Mcr,
399 curves.back ());
400 i += 3 + number_of_points * 3; // 3 - time,align,number_of_points,points...
401 continue;
402 }
403 case SCALE:
404 glScalef (operations->at (i + 1),
405 operations->at (i + 2),
406 operations->at (i + 3));
407 i += 3;
408 continue;
409 case BEGIN_GROUP:
410 glPushMatrix ();
411 continue;
412 case END_GROUP:
413 glPopMatrix ();
414 continue;
415 case LOAD_MODEL:
416 {
417 int stringSize = (int) operations->at (i + 1);
418 if (!hasPushedModels)
419 {
420 char modelName[stringSize + 1];
421
422 int j;
423 for (j = 0; j < stringSize; j++)
424 modelName[j] = (char) operations->at (i + 2 + j);
425
426 modelName[j] = 0;
427
428 globalModels.push_back (allocModel (modelName));
429 glGenBuffers (1, &vbo_indices[model_num]);
430 glBindBuffer (GL_ARRAY_BUFFER, vbo_indices[model_num]);
431 Model m = globalModels.back ();
432 glBufferData (GL_ARRAY_BUFFER, sizeof (float) * 3 * m->nVertices, m->vertices, GL_STATIC_DRAW);
433 glBindBuffer (GL_ARRAY_BUFFER, 0); //unbind
434 free (m->vertices);
435 }
436
437 renderModel (globalModels[model_num], vbo_indices[model_num]);
438 model_num++;
439 i += stringSize + 1; //just to be explicit
440 continue;
441 }
442 }
443 }
444 hasPushedModels = true;
445 hasLoadedCurves = true;
446}

494int timebase = 0, frame = 0;
495void renderScene ()
496{
497 float fps;
498 int time;
499 char s[64];
500
501 // clear buffers
502 glClear (GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
503
504 // helps in avoiding rounding mistakes, discards old matrix, read more
505 glLoadIdentity ();
506
507 /*draw absolute (before any transformation) axes*/
508 draw_axes ();
509
510 // set the camera
511 gluLookAt (globalEyeX, globalEyeY, globalEyeZ,
512 globalCenterX, globalCenterY, globalCenterZ,
513 globalUpX, globalUpY, globalUpZ);
514
515 // put the geometric transformations here
516 glRotatef (globalAngle, globalRotateX, globalRotateY, globalRotateZ);
517 glTranslatef (globalTranslateX, globalTranslateY, globalTranslateZ);
518 glScalef (globalScaleX, globalScaleY, globalScaleZ);
519
520 operations_render (&globalOperations);
521
522 frame++;
523 time = glutGet (GLUT_ELAPSED_TIME);
524 if (time - timebase > 1000)
525 {
526 fps = frame * 1000.0 / (time - timebase);
527 timebase = time;
528 frame = 0;
529 sprintf (s, "FPS: %f6.2", fps);
530 glutSetWindowTitle (s);
531 }
532
533 // End of frame
534 redisplay ();
535 glutSwapBuffers ();
536}

546void keyboardFunc (unsigned char key, int xmouse, int ymouse)
547{
548 /*
549 Rotation:
550 x y z x y z
551 X Y Z x y z
552 angle:
553 , . -=s +=s
554 < > s s
555 translation:
556 q w e y z y
557 a s d x z x
558 scaling:
559 u i o y z y
560 j k l x z x
561 background color: r g b R G B
562 camera:
563 1 2 3 ! @ # EyeX EyeY EyeZ
564 4 5 6 $ % ^ CenterX CenterY CenterZ
565 7 8 9 & * ( UpX UpY UpZ
566 */
567 switch (key)
568 {
569 /*Rotation*/
570 case 'x':
571 globalRotateX -= 1;
572 break;
573
574 case 'X':
