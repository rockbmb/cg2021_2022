211static inline void
212model_cone_vertex (const float r, const float height, const float theta, const float h, unsigned int *pos,
213 float *points)
214{
215 /*
216 x = r ⋅ (h/height) ⋅ cos(θ)
217 y = 2 ⋅ (height + h)
218 z = r ⋅ (h/height) ⋅ sin(θ)
219
220 r ≥ 0
221 θ ∈ {-π + i⋅s : s = 2π/slices ∧ i ∈ {0,...,slices} }
222 h ∈ {-height + j⋅t : t = height/stacks ∧ j ∈ {0,...,stacks} }
223
224 check:
225 1. https://www.math3d.org/7oeSkmuns
226 */
227 points_vertex (r * h / height * cos (theta), height + h, r * h / height * sin (theta), pos, points);
228}

284model_sphere_vertex (const float r, const float theta, const float phi, unsigned int *pos, float *points)
285{
286 /*
287 x = r ⋅ sin(θ)cos(φ)
288 y = r ⋅ sin(φ)
289 z = r ⋅ cos(θ)cos(φ)
290
291 r ≥ 0
292 θ ∈ {-π + i⋅s : s = 2π/slices ∧ i ∈ {0,...,slices} }
293 ϕ ∈ {-π/2 + j⋅t : t = π/stacks ∧ j ∈ {0,...,stacks} }
294
295 check
296 1. https://www.math3d.org/EumEEZBKe
297 2. https://www.math3d.org/zE4n6xayX
298 */
299 points_vertex (r * sin (theta) * cos (phi), r * sin (phi), r * cos (theta) * cos (phi), pos, points);

338vector<array<vec3, 16>> read_Bezier (const char *patch)
339{
340 string buffer;
341 ifstream myFile;
342
343 myFile.open (patch, ios::in | ios::out);
344 getline (myFile, buffer);
345 // Número de patches presentes no ficheiro.
346 int n_patches = stoi (buffer);
347
348 // Vetor de vetores de índices.
349 vector<vector<int>> patches;
350
351 // Ciclo externo lê uma linha (patch) de cada vez
352 for (int j = 0; j < n_patches; j++)
353 {
354 vector<int> patchIndexes;
355 /*
356 Ciclo interno lê os índices dos pontos de controlo de cada patch, sabendo que cada
357 patch terá 16 pontos de controlo.
358 */
359 for (int i = 0; i < 15; i++)
360 {
361 getline (myFile, buffer, ',');
362 patchIndexes.push_back (stoi (buffer));
363 }
364 getline (myFile, buffer);
365 patchIndexes.push_back (stoi (buffer));
366 patches.push_back (patchIndexes);
367 }
368
369 getline (myFile, buffer);
370 // Número de pontos presentes no ficheiro.
371 int pts = stoi (buffer);
372
373 // Vetor que guardará as coordenadas de pontos de controlo para superfície de Bézier.
374 vector<vec3> control;
375 for (int j = 0; j < pts; j++)
376 {
377 vec3 v;
378 getline (myFile, buffer, ',');
379 v[0] = stof (buffer);
380 getline (myFile, buffer, ',');
381 v[1] = stof (buffer);
382 getline (myFile, buffer);
383 v[2] = stof (buffer);
384 control.push_back (v);
385 }
386
387 /*
388 Percorrem-se os vetores que, para cada patch, guardam os seus índices de pontos de controlo.
389 Para cada patch, constroi-se um vetor com as coordenadas dos seus pontos de controlo.
390 */
391 vector<array<vec3, 16>> pointsInPatches;
392 for (auto &patche : patches)
393 {
394 array<vec3, 16> pointsInPatch{};
395 for (int j = 0; j < 16; j++)
396 {
397 pointsInPatch[j] = control[patche[j]];
398 }
399 pointsInPatches.push_back (pointsInPatch);
400 }
401
402 myFile.close ();
403
404 /*std::cout << "read_Bezier read:" << std::endl;
405 for (auto arr : pointsInPatches)
406 for (auto p : arr)
407 std::cout << glm::to_string (p) << std::endl;*/
408 return pointsInPatches;
409}
