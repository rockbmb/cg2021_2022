33void mult (const T1 &a, const T2 &b, T3 &r, const int m, const int p, const int q)
34{
35 for (auto i = 0; i < m; ++i)
36 {
37 for (auto j = 0; j < q; ++j)
38 {
39 r[m * i + j] *= 0;
40 for (auto k = 0; k < p; k++)
41 {
42 r[m * i + j] += a[m * i + k] * b[p * k + j];
43 }
44 }
45 }

48void get_curve (const float time, const mat4 &M, const array<vec3, 4> &control_points, vec3 &pos, vec3 &deriv)
49{
50 const auto t = glm::vec4 (pow (time, 3), pow (time, 2), time, 1);
51 const auto t_prime = glm::vec4 (3 * pow (time, 2), 2 * time, 1, 0);
52 mat4x3 K (control_points[0], control_points[1], control_points[2], control_points[3]);
53
54 const auto KM = K * M;
55 pos = KM * t;
56 deriv = KM * t_prime;
