119void changeSize(int w, int h) {
120
121 // Prevent a divide by zero, when window is too short
122 // (you cant make a window with zero width).
123 if (h == 0)
124 h = 1;
125
126 // compute window's aspect ratio
127 float ratio = w * 1.0 / h;
128
129 // Set the projection matrix as current
130 glMatrixMode(GL_PROJECTION);
131 // Load Identity Matrix
132 glLoadIdentity();
133
134 // Set the viewport to be the entire window
135 glViewport(0, 0, w, h);
136
137 // Set perspective
138 gluPerspective(globalFOV, ratio, globalNear, globalFar);
139
140 // return to the model view matrix mode
141 glMatrixMode(GL_MODELVIEW);
142}

148void drawCylinder(float r, float height, int slices) {
149 /*coloring*/
150 float c = 1;
151 float color_step = (float) 1.0 / (float) (slices + 4);
152
153 float s = 2.0f * (float) M_PI / (float) slices;
154 float theta = -M_PI;
155
156 glBegin(GL_TRIANGLES);
157 for (int m = 1; m <= slices; m++) {
158 float i = (float) m;
159
160 //base
161 glColor3f(0, 0, c);
162 glVertex3d(0, 0, 0); //O
163 cylinderVertex(r, 0, theta + s * i); //P2
164 cylinderVertex(r, 0, theta + s * (i - 1)); //P1
165
166 //top
167 glColor3f(0, c, 0);
168 glVertex3d(0, height, 0); //O'
169 cylinderVertex(r, height, theta + s * (i - 1)); //P1'
170 cylinderVertex(r, height, theta + s * i); //P2'
171
172 // midsection
173 glColor3f(c, 0, 0);
174 cylinderVertex(r, 0, theta + s * (i - 1)); //P1
175 cylinderVertex(r, 0, theta + s * i); // P2
176 cylinderVertex(r, height, theta + s * (i - 1)); // P1'
177
178 glColor3f(c, 0, c);
179 cylinderVertex(r, height, theta + s * i); // P2'
180 cylinderVertex(r, height, theta + s * (i - 1)); // P1'
181 cylinderVertex(r, 0, theta + s * i); //P2
182
183 c -= color_step;
184 }
185 glEnd();
186}

192void drawCone(float r, float height, unsigned int slices, unsigned int stacks) {
193 /*coloring*/
194 float c1 = 1;
195 float c2 = 1;
196 float color_step = (float) 1.0 / (float) (slices + 4);
197
198 // https://www.math3d.org/5gLCN9yBz
199
200 float s = 2.0f * (float) M_PI / (float) slices;
201 float t = height / (float) stacks;
202 float theta = -M_PI;
203 float h = -height;
204
205 glBegin(GL_TRIANGLES);
206 for (int m = 1; m <= slices; m++) {
207 for (int n = 1; n <= stacks; n++) {
208 float i = (float) m;
209 float j = (float) n;
210
211 //base
212 glColor3f(c1, c2, c2);
213 glVertex3d(0, 0, 0); //O
214 coneVertex(r, height, theta + s * (i - 1), h); //P1
215 coneVertex(r, height, theta + s * i, h); //P2
216
217 glColor3f(c1, 0, c2);
218 coneVertex(r, height, theta + s * i, h + t * (j - 1)); // P2
219 coneVertex(r, height, theta + s * (i - 1), h + t * j); // P1'
220 coneVertex(r, height, theta + s * i, h + t * j); //P2'
221
222 glColor3f(c1, c2, 0);
223 coneVertex(r, height, theta + s * i, h + t * (j - 1)); // P2
224 coneVertex(r, height, theta + s * (i - 1), h + t * (j - 1)); // P1
225 coneVertex(r, height, theta + s * (i - 1), h + t * j); // P1'
226
227 c2 -= color_step;
228 }
229 c1 -= color_step;
230 c2 = 1;
231 }
232 glEnd();
233}

457void keyboardFunc(unsigned char key, int xmouse, int ymouse) {
458/*
459 Rotation:
460 x y z x y z
461 X Y Z x y z
462 angle:
463 , . -=s +=s
464 < > s s
465 translation:
466 q w e y z y
467 a s d x z x
468 scaling:
469 u i o y z y
470 j k l x z x
471 background color: r g b R G B
472 camera:
473 1 2 3 ! @ # EyeX EyeY EyeZ
474 4 5 6 $ % ^ CenterX CenterY CenterZ
475 7 8 9 & * ( UpX UpY UpZ
476 */
477 switch (key) {
478 case 'x':
479 globalRotateX -= 1;
480 break;
481
482 case 'X':
483 globalRotateX += 1;
484 break;
485
486 case 'y':
487 globalRotateY -= 1;
488 break;
489
490 case 'Y':
491 globalRotateY += 1;
492 break;
493
494 case 'z':
495 globalRotateZ -= 1;
496 break;
497
498 case 'Z':
499 globalRotateZ += 1;
500 break;
501
502 case ',':
503 globalAngle -= globalAngleStep;
504 break;
505
506 case '.':
507 globalAngle += globalAngleStep;
508 break;
509
510 case '<':
511 globalAngleStep /= 2;
512 break;
513
514 case '>':
515 globalAngleStep *= 2;
516 break;
517
518 /*translation*/
519
520 /*z-axis*/
521 case 'w':
522 globalTranslateZ += globalTranslateStep;
523 break;
524 case 's':
525 globalTranslateZ -= globalTranslateStep;
526 break;
527
528 /*x-axis*/
529 case 'a':
530 globalTranslateX -= globalTranslateStep;
531 break;
532 case 'd':

534 break;
535
536 /*y-axis*/
537 case 'q':

539 break;
540 case 'e':

542 break;
543 /*end of translation*/
544
545 /*scaling*/

549 globalScaleZ += globalScaleStep;

551 case 'k':
552 globalScaleZ -= globalScaleStep;
553 break;
554
555 /*x-axis*/
556 case 'j':
557 globalScaleX -= globalScaleStep;
558 break;
559 case 'l':
560 globalScaleX += globalScaleStep;

571

576 case 'R':

578 break;
579 case 'g':
580 globalBackgroundColorG -= globalBackgroundColorStep;
581 break;
582 case 'G':

584 break;

595 break;

598 break;
599 case '2':
600 globalEyeY -= globalEyeStep;
